Notes from Javascript.info

Object Methods

Object:

let user = {

name: "John",

age: 30

};

Method

let user = {

name: "John",

age: 30

};

user.sayHi = function() {

alert("Hello!");

};

user.sayHi(); // Hello!

Method shorthand

// these objects do the same

user = {

sayHi: function() {

alert("Hello");

}

};

// method shorthand looks better, right?

user = {

sayHi() { // same as "sayHi: function(){...}"

alert("Hello");

}

};

“this” in methods

let user = {

name: "John",

age: 30,

sayHi() {

// "this" is the "current object"

alert(this.name);

}

};

user.sayHi(); // John
